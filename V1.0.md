## 四、数据库设计

### 4.1 新增表

```sql
-- 用户表
CREATE TABLE users (
    id VARCHAR(64) PRIMARY KEY,
    username VARCHAR(128) UNIQUE,
    email VARCHAR(256),
    password_hash VARCHAR(256),
    created_at BIGINT,
    updated_at BIGINT
);

-- 群组表
CREATE TABLE groups (
    group_id VARCHAR(64) PRIMARY KEY,
    name VARCHAR(256),
    owner VARCHAR(64),
    created_at BIGINT
);

-- 群成员表
CREATE TABLE group_members (
    group_id VARCHAR(64),
    user_id VARCHAR(64),
    role VARCHAR(32),  -- owner, admin, member
    joined_at BIGINT,
    PRIMARY KEY (group_id, user_id)
);

-- 角色表
CREATE TABLE roles (
    role_id VARCHAR(64) PRIMARY KEY,
    name VARCHAR(64),
    permissions JSON
);

-- 用户角色表
CREATE TABLE user_roles (
    user_id VARCHAR(64),
    role_id VARCHAR(64),
    PRIMARY KEY (user_id, role_id)
);
```
